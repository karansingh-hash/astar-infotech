   110→/* ── Animated Section Wrapper ── */
   111→function AnimatedSection({ children, className = '', id }: { children: React.ReactNode; className?: string; id?: string }) {
   112→  const ref = useRef(null)
   113→  const isInView = useInView(ref, { once: true, margin: '-80px' })
   114→  return (
   115→    <section id={id} ref={ref} className={className}>
   116→      <motion.div initial={{ opacity: 0, y: 40 }} animate={isInView ? { opacity: 1, y: 0 } : { opacity: 0, y: 40 }} transition={{ duration: 0.6, ease: 'easeOut' }}>
   117→        {children}
   118→      </motion.div>
   119→    </section>
   120→  )
   121→}

   175→  const [portfolioForm, setPortfolioForm] = useState({ title: '', category: '', description: '', tech: '', color: '', image: '', order: 0 })

   183→  useEffect(() => { if (externalOpen) setIsOpen(true) }, [externalOpen])
   184→  useEffect(() => {
   185→    if (!isOpen || !isAuthenticated) return
   186→    if (activeTab === 'dashboard') fetchDashboard()
   187→    if (activeTab === 'inquiries') fetchContacts()
   188→    if (activeTab === 'services') fetchServices()

   190→    if (activeTab === 'testimonials') fetchTestimonials()
   191→    if (activeTab === 'statistics') { fetchDashboard(); fetchStats() }

   194→
   195→  const handleLogin = async (e: FormEvent) => {
   196→    e.preventDefault(); if (!password.trim()) return; setAuthLoading(true)
   197→    try {
   198→      const res = await fetch('/api/admin/auth', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ password }) })

   201→      else toast.error('Login Failed', { description: data.error || 'Invalid password.' })
   202→    } catch { toast.error('Error', { description: 'Failed to connect to server.' }) } finally { setAuthLoading(false) }
   203→  }

   207→  const fetchContacts = async () => { setContactsLoading(true); try { const r = await fetch('/api/contacts'); const d = await r.json(); setContacts(d.contacts || []) } catch { toast.error('Error', { description: 'Failed to fetch contacts.' }) } finally { setContactsLoading(false) } }
   208→  const fetchServices = async () => { setServicesLoading(true); try { const r = await fetch('/api/services'); const d = await r.json(); setServices(d.services || []) } catch { toast.error('Error', { description: 'Failed to fetch services.' }) } finally { setServicesLoading(false) } }
